    int solveSudoku(vector<vector<char>>& board) {
        for (int row = 0; row < N; row++) {
            for (int col = 0; col < N; col++) {
                if (board[row][col] == '$') {
                    int count = 0;
                    for (int i = 1; i <= 9; i++) {
                        if (isValid(row, col, i, board)) {
                            board[row][col] = i + '0';
                            count += solveSudoku(board);
                            board[row][col] = '$';  // »ØËÝ
                        }
                    }
                    return count;
                }
            }
        }

        return 1;  // Êý¶ÀÒÑ¾­ÌîÂú£¬·µ»Ø 1
